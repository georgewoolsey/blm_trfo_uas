5.1 Group Tree Crowns

While some research on tree spatial patterns use tree point locations buffered with a fixed radius (e.g. 3 m) to approximate the crown radius of a “typical” overstory tree (e.g. Churchill et al. (2016)), we’ll use the irregular (more realistic?) crown shapes identified from the aerial data to determine tree clump groups.

Remember we are only looking at what could be considered “overstory” trees since we filtered the tree list to keep only trees that meet a certain minimum height

Since a primary objective of this analysis is to evaluate how different treatments influence forest structure, we’ll treat each unit as a discrete area and exclude any trees located outside of the unit boundaries from the clump classification process. This approach assumes that trees within a unit do not form functional clusters with those in adjacent areas (even though some units share a common border and we have data extending beyond the unit boundaries). This assumption prevents data from untreated control areas or differently treated areas from influencing the results when summarized by treatment type. This intentional ignorance of edge effects means that the scale of focus of this analysis is strictly on the treatment unit.

get_clumps_temp <- function(tree_list) {
  # buffer and dissolve (arcGIS logic)
  clump_polys <- tree_list %>%
    dplyr::ungroup() %>% 
    sf::st_union() %>%
    sf::st_cast("POLYGON") %>%
    sf::st_as_sf() %>%
    dplyr::mutate(clump_id = dplyr::row_number())

  # attach to trees
  trees_with_clumps <- tree_list %>%
    # get rid of columns we'll create
    dplyr::select( -dplyr::any_of(c(
      "hey_xxxxxxxxxx"
      , "clump_id"
      , "clump_n_trees"
      , "clump_n_trees_grp"
      , "clump_dist_m"
    ))) %>%
    sf::st_join(clump_polys %>% dplyr::select(clump_id), join = sf::st_intersects, largest = T) %>%
    dplyr::group_by(clump_id) %>%
    dplyr::mutate(
      clump_n_trees = dplyr::n()
      , clump_n_trees_grp = dplyr::case_when(
        clump_n_trees == 1 ~ "Individual"
        , clump_n_trees >= 2 & clump_n_trees <= 4 ~ "2-4 trees"
        , clump_n_trees >= 5 & clump_n_trees <= 9 ~ "5-9 trees"
        , clump_n_trees >= 10 & clump_n_trees <= 15 ~ "10-15 trees"
        , clump_n_trees > 15 ~ ">15 trees"
        # , T ~ "error"
      ) %>%
      factor(
        ordered = T
        , levels = c("Individual", "2-4 trees", "5-9 trees", "10-15 trees", ">15 trees")
      )
    ) %>%
    dplyr::ungroup()
  # add back to clumps
  clump_polys <- clump_polys %>% 
    dplyr::inner_join(
      trees_with_clumps %>% 
        sf::st_drop_geometry() %>% 
        dplyr::distinct(clump_id, clump_n_trees, clump_n_trees_grp)
      , by = "clump_id"
    )
  return(list(clump_polys = clump_polys, tree_list = trees_with_clumps))
}

5.2 Identify Openings

Let’s identify the gaps within a treatment unit. We’ll identify openings based on their distance from the nearest tree crown where an opening is any area located at least 3 m from the nearest tree (Churchill et al. 2016). This 3 m buffer accounts for the zone of immediate canopy influence and ensures that areas categorized as open are functionally distinct from the canopy. This threshold is flexible and can be adjusted to meet specific management or analysis objectives (e.g. flame length; wildlife corridors).

First, we’ll generate a distance raster where each cell represents the distance to the nearest tree clump group (i.e. crown) within the treatment unit boundary. Areas with a distance value of 3 m or greater are classified as openings and are defined by spatial connectedness, not a specific geometric shape. As such, they may be irregularly shaped as they are allowed to wind between clumps to follow a path where trees are distant. A single opening continues as long as the flow of open space is not entirely obstructed by surrounding tree groups. Once a gap is completely surrounded by canopy, the opening is closed off and is treated as a discrete spatial feature.

# create function to pass a return from get_clump_summary() and get a distance raster
get_clump_dist_rast_temp <- function(x, clump_dist_m = 3){
  # get clump_dist_m
  # clump_dist_m <- min(get_clump_summary_ans$clump_dist_m, na.rm = T)
  # rasterize the clump polygons and then calculate distance between clumps as raster
  dist_rast <-
    terra::rasterize(
        x = 
          crowns_sf %>% 
          dplyr::inner_join(
            unq_units_df %>% dplyr::slice(x)
            , by = dplyr::join_by(trtmnt_type,trtmnt_block,trtmnt_unit)
          ) %>% 
          terra::vect()
        , y = 
          # crowns_sf %>% 
          # dplyr::inner_join(
          #   unq_units_df %>% dplyr::slice(x)
          #   , by = dplyr::join_by(trtmnt_type,trtmnt_block,trtmnt_unit)
          # ) %>% 
          # terra::vect() %>%
          stand_boundary %>% 
          dplyr::filter(
            unit == unq_units_df$trtmnt_unit[x]
            , block == unq_units_df$trtmnt_block[x]
            , stringr::str_squish(tolower(tretmnt)) == unq_units_df$trtmnt_type[x]
          ) %>% 
          terra::vect() %>% 
          terra::rast(res = 0.2)
      ) %>% 
      terra::distance() %>% 
      # crop it to stand extent
      terra::crop(
        stand_boundary %>% 
          dplyr::filter(
            unit == unq_units_df$trtmnt_unit[x]
            , block == unq_units_df$trtmnt_block[x]
            , stringr::str_squish(tolower(tretmnt)) == unq_units_df$trtmnt_type[x]
          ) %>% 
          terra::vect()
        , mask = T
      )
  ######### part 2
  # now create openings vector data
  openings_vect <- 
    dist_rast %>% 
      terra::classify(rcl = c(clump_dist_m,Inf), others = NA, include.lowest = T) %>% 
      terra::as.polygons(na.rm = T) %>% 
      sf::st_as_sf() %>% 
      sf::st_cast("POLYGON") %>% 
      dplyr::mutate(layer = dplyr::row_number()) %>% 
      dplyr::mutate(
        openining_area_m2 = sf::st_area(geometry) %>% as.numeric()
        , clump_dist_m = clump_dist_m
      )
  
  # return
  return(list(dist_rast = dist_rast, openings_vect = openings_vect))
}

# create a function to summarize by number of tree clump grouping
get_clump_n_trees_grp_summary_temp = function(x){
  # get area of harvest unit
  #...will use this area in the area calculations such that...
  #...TPA = trees in a certain group size across the whole stand area
  harvest_area_ha <- 
    stand_boundary %>% 
    dplyr::filter(
      unit == unq_units_df$trtmnt_unit[x]
      , block == unq_units_df$trtmnt_block[x]
      , stringr::str_squish(tolower(tretmnt)) == unq_units_df$trtmnt_type[x]
    ) %>% 
    sf::st_transform(terra::crs(chm_rast)) %>% 
    sf::st_union() %>% 
    sf::st_as_sf() %>% 
    dplyr::mutate(harvest_area_ha = sf::st_area(.) %>% as.numeric() %>% `/`(10000)) %>% 
    dplyr::pull(harvest_area_ha) %>% 
    .[1]
  # which dta
  get_tree_clumps_ans <- 
    crowns_sf %>% 
    sf::st_drop_geometry() %>% 
    dplyr::ungroup() %>% 
    dplyr::inner_join(
      unq_units_df %>% dplyr::slice(x)
      , by = dplyr::join_by(trtmnt_type,trtmnt_block,trtmnt_unit)
    ) %>% 
    dplyr::mutate(stand_area_ha = harvest_area_ha)
  # collapse and calculate silv metrics
  dta <- 
    get_tree_clumps_ans %>% 
    dplyr::group_by(trtmnt_type,trtmnt_block,trtmnt_unit,stand_area_ha,clump_n_trees_grp) %>%
    dplyr::summarise(
      # summary metrics
      n_trees = dplyr::n_distinct(treeID)
      , mean_dbh_cm = mean(dbh_cm, na.rm = T)
      , mean_tree_height_m = mean(tree_height_m, na.rm = T)
      , loreys_height_m = sum(basal_area_m2*tree_height_m, na.rm = T) / sum(basal_area_m2, na.rm = T)
      , basal_area_m2 = sum(basal_area_m2, na.rm = T)
      , sum_dbh_cm_sq = sum(dbh_cm^2, na.rm = T)
      , .groups = "drop_last"
    ) %>%
    dplyr::ungroup() %>% 
    dplyr::mutate(
      trees_per_ha = (n_trees/stand_area_ha) # (n_trees/clump_area_ha) ... this was not right
      , basal_area_m2_per_ha = (basal_area_m2/stand_area_ha) # (basal_area_m2/clump_area_ha) ... this was not right
      , qmd_cm = sqrt(sum_dbh_cm_sq/n_trees)
      # stand calcs
      , stand_trees_per_ha = sum(n_trees)/stand_area_ha
      , stand_basal_area_m2 = sum(basal_area_m2)
      , stand_basal_area_m2_per_ha = sum(basal_area_m2)/stand_area_ha
      , pct_stand_basal_area = basal_area_m2/stand_basal_area_m2
      , pct_stand_n_trees = n_trees/sum(n_trees)
      , stand_qmd_cm = sqrt(sum(get_tree_clumps_ans$dbh_cm^2, na.rm = T)/sum(n_trees))
    ) %>%
    dplyr::select(-c(sum_dbh_cm_sq)) %>% 
    # convert to imperial units
    calc_imperial_units_fn()
  # fill in missing
  ret_dta <- 
    dplyr::tibble(
      # get all possible labels from the factor levels
      clump_n_trees_grp = levels(crowns_sf$clump_n_trees_grp)
    ) %>% 
    dplyr::mutate(
      dplyr::across(
        dplyr::all_of(c("clump_n_trees_grp"))
        , ~ factor(
          .x
          , levels = levels(crowns_sf[[dplyr::cur_column()]])
          , ordered = T
        )
      )
    ) %>% 
    dplyr::bind_cols(unq_units_df %>% dplyr::slice(x)) %>% 
    dplyr::left_join(
      dta
      , by = dplyr::join_by(trtmnt_type,trtmnt_block,trtmnt_unit,clump_n_trees_grp)
    ) %>% 
    dplyr::mutate(
      dplyr::across(
        tidyselect::starts_with("stand_")
        , ~ max(.x, na.rm = T)
      )
      , dplyr::across(
        dplyr::where(is.numeric)
        , ~ dplyr::coalesce(.x, 0)
      )
    )
  # return
  return(ret_dta)
}
